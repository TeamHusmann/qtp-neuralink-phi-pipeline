function getBandForBracket(n) {

function buildBracketShells() {
  // Render every 3rd bracket for performance (still 98 shells)
  for (let n = 3; n <= BRACKET_COUNT; n += 3) {
    const band = getBandForBracket(n);
    const bandData = SIGMA_BANDS[band];

    // Pure geodesics (κ=0) get special treatment — brighter, golden color
    const isGeodesic = kappa === 0;
    const color = isGeodesic ? 0xffd700 : bandData.color;
    const emissiveIntensity = isGeodesic ? 0.4 : 0.08;

    const material = new THREE.MeshPhongMaterial({
      color,
      emissive: color,
      emissiveIntensity,
      transparent: true,
      opacity: 0.12,
      wireframe: true,
      depthWrite: false,
    });

    const mesh = new THREE.Mesh(geometry, material);
    mesh.userData.bracketIndex = n;
    mesh.userData.band = band;
    mesh.userData.isGeodesic = isGeodesic;
    mesh.userData.kappa = kappa;
    mesh.userData.baseRadius = visualRadius;
    scene.add(mesh);
    shells[n] = mesh;
  }

  // Key brackets always rendered with full brightness
  for (const [name, n] of Object.entries(KEY_BRACKETS)) {
    if (shells[n]) {
      shells[n].material.opacity = 0.35;
      shells[n].material.emissiveIntensity = 0.6;
    }
  }
}

  scene.add(hingePortal);

function buildBackbone() {
  // Fibonacci number positions as glowing particles
  const fibs = [];
  let a = 1, b = 1;
  while (b <= N_HUBBLE * 3) { fibs.push(b); [a, b] = [b, a + b]; }

  const positions = [];
  const goldenAngle = GOLDEN_ANGLE * Math.PI / 180;

  fibs.forEach((f, i) => {
    const r = (Math.min(f, N_HUBBLE) / N_HUBBLE) * MAX_VISUAL_RADIUS * 0.9;
    const theta = i * goldenAngle;
    const phi = Math.acos(1 - 2 * (i + 0.5) / fibs.length);
    positions.push(
      r * Math.sin(phi) * Math.cos(theta),
      r * Math.sin(phi) * Math.sin(theta),
      r * Math.cos(phi)
    );
  });

  const geometry = new THREE.BufferGeometry();
  geometry.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3));
  const material = new THREE.PointsMaterial({ color: 0xffd700, size: 3, transparent: true, opacity: 0.8 });
  backboneParticles = new THREE.Points(geometry, material);
  scene.add(backboneParticles);
}

function buildStarfield() {

  for (let i = 0; i < 8000; i++) {
    positions.push(
      (Math.random() - 0.5) * 4000,

  geo.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3));
  scene.add(new THREE.Points(geo, new THREE.PointsMaterial({ color: 0xffffff, size: 0.5, transparent: true, opacity: 0.4 })));
}

export function updateAllBrackets(t) {

    const ni = parseInt(n);

// Decode a bracket address and return full analysis
export function decodeAddress(m) {
  const terms = zeckendorf(m);
  const kappa = terms.length - 1;
  const scale = bracketScale(Math.min(m, N_HUBBLE));
  const isPureGeodesic = kappa === 0;

  // Find the nearest key bracket
  let nearestKey = null, nearestDist = Infinity;
  for (const [name, n] of Object.entries(KEY_BRACKETS)) {
    const d = Math.abs(m - n);
    if (d < nearestDist) { nearestDist = d; nearestKey = name; }
  }

  return {
    address: m,
    zeckendorf: terms,
    kappa,
    isPureGeodesic,
    scale,
    nearestKey,
    nearestDist,
    energySector: getEnergySector(m),
  };
}

function getEnergySector(n) {

}

import * as THREE from 'https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js';
import { initScene, updateAllBrackets, rotateShells, scene, camera, renderer, shells } from './universe.js';

// dat.GUI
const gui = new dat.GUI({ width: 300 });
initUI(gui);

// Three.js OrbitControls (manual implementation — r128 doesn't auto-bundle)
let isDragging = false, prevMouse = { x: 0, y: 0 };
let cameraTheta = 0, cameraPhi = Math.PI / 4, cameraRadius = 800;

function updateCamera() {
  camera.position.set(
    cameraRadius * Math.sin(cameraPhi) * Math.sin(cameraTheta),
    cameraRadius * Math.cos(cameraPhi),
    cameraRadius * Math.sin(cameraPhi) * Math.cos(cameraTheta)
  );
  camera.lookAt(0, 0, 0);
}

document.addEventListener('mousedown', e => { isDragging = true; prevMouse = { x: e.clientX, y: e.clientY }; });
document.addEventListener('mouseup',   () => { isDragging = false; });
document.addEventListener('mousemove', e => {
  if (!isDragging) return;
  const dx = e.clientX - prevMouse.x;
  const dy = e.clientY - prevMouse.y;
  cameraTheta -= dx * 0.005;
  cameraPhi = Math.max(0.1, Math.min(Math.PI - 0.1, cameraPhi + dy * 0.005));
  prevMouse = { x: e.clientX, y: e.clientY };
  updateCamera();
});
document.addEventListener('wheel', e => {
  cameraRadius = Math.max(100, Math.min(1500, cameraRadius + e.deltaY * 0.5));
  updateCamera();
});

// Init scene
const container = document.getElementById('app');
initScene(container);

// HUD
const hud = buildHUD(container);

let lastTime = 0;
let cosmicTime = 0;

function animate(timestamp) {
  requestAnimationFrame(animate);
  const delta = Math.min(timestamp - lastTime, 50); // cap at 50ms
  lastTime = timestamp;

  // Advance cosmic time if playing
  if (params.autoTimeline) {
    cosmicTime = Math.min(1.0, cosmicTime + (delta / 1000) * params.timeSpeed * 10);
    params.cosmicTime = cosmicTime;
    if (cosmicTime >= 1.0) params.autoTimeline = false;
  } else {
    cosmicTime = params.cosmicTime;
  }

  // Update bracket shells
  updateAllBrackets(cosmicTime);

  // Rotate
  rotateShells(delta * params.rotationSpeed);

  // Order parameter
  const orderParam = computeOrderParameter(cosmicTime);

  // HUD
  if (params.showHUD) updateHUD(hud, cosmicTime, orderParam);

  renderer.render(scene, camera);
}

requestAnimationFrame(animate);
