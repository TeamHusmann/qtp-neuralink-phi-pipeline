## THE ZECKENDORF DECODER

Every cycle count maps to a unique set of active brackets via Zeckendorf decomposition.
Implement this exactly — it is the heart of the visual model.

```javascript
// Generate Fibonacci sequence up to needed maximum
function buildFibonacci(maxN) {
  const F = [0, 1];
  while (F[F.length - 1] < maxN) {
    F.push(F[F.length - 1] + F[F.length - 2]);
  }
  return F;
}

const FIBONACCI = buildFibonacci(1000);

// Zeckendorf decomposition — unique representation as sum of non-consecutive Fibonacci numbers
function zeckendorf(n) {
  const indices = [];
  let remaining = Math.floor(n);
  let k = FIBONACCI.length - 1;
  while (FIBONACCI[k] > remaining) k--;
  while (remaining > 0 && k >= 2) {
    if (FIBONACCI[k] <= remaining) {
      indices.push(k);
      remaining -= FIBONACCI[k];
      k -= 2;
    } else {
      k--;
    }
  }
  return indices.sort((a, b) => a - b);
}

// Get Fibonacci values from indices
function zeckendorfValues(n) {
  return zeckendorf(n).map(i => FIBONACCI[i]);
}
```

```javascript
function fibonacciHelix(n_turns, points_per_turn, radius, pitch) {
  const points = [];
  const total = n_turns * points_per_turn;
  for (let i = 0; i < total; i++) {
```
